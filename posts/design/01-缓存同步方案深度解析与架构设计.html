    private static final long DEFAULT_EXPIRE_TIME = 30000; // 30ç§’

     * è·å–åˆ†å¸ƒå¼é”

    public boolean tryLockWithRetry(String key, long expireTime, int retryTimes, long sleepTime) {

            }

                Thread.sleep(sleepTime);

                Thread.currentThread().interrupt();
                return false;
            }
        }
        
        return false;
